!/usr/bin/env python3
"""
安全终端命令执行器
仅允许白名单命令
限制工作目录
超时控制
"""
import subprocess
import sys
import os
import re

安全配置
ALLOWED_COMMANDS = {
    'ls', 'pwd', 'cat', 'echo', 'grep', 'find', 'head', 'tail',
    'python3', 'node', 'git', 'curl', 'wget', 'df', 'free'
}
WORKSPACE = "/home/lz/.openclaw/workspace"
TIMEOUT = 30  # 秒

def sanitize_command(cmd: str) -> str:
    """清理命令，禁止危险字符"""
    # 禁止管道/重定向/分号等
    if any(c in cmd for c in [';', '&', '|', '', '', '>', ''", file=sys.stderr)
        sys.exit(1)
    
    command = sys.argv[1]
    
    try:
        safe_cmd = sanitize_command(command)
        result = subprocess.run(
            safe_cmd,
            shell=True,
            cwd=WORKSPACE,
            capture_output=True,
            text=True,
            timeout=TIMEOUT
        )
        output = result.stdout or result.stderr
        print(output[:2000])  # 限制输出长度
        
    except subprocess.TimeoutExpired:
        print("❌ 命令执行超时", file=sys.stderr)
        sys.exit(1)
    except Exception as e:
        print(f"❌ 执行错误: {e}", file=sys.stderr)
        sys.exit(1)

if name == "main":
    main()
