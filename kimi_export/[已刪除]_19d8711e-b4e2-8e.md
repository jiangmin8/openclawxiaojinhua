## 🔧 完整硬盘迁移脚本

#!/bin/bash

# ============================================
# GL-MT3600BE 全盘迁移到 USB 硬盘脚本
# 硬盘挂载点：/mnt/sda4（根据实际情况修改）
# ============================================

# 1. 确认硬盘挂载点
DISK="/mnt/sda4"
echo "=== 检查硬盘挂载 ==="
df -h | grep sda
ls -la $DISK

# 2. 创建目录结构
echo "=== 创建硬盘目录结构 ==="
mkdir -p $DISK/overlay
mkdir -p $DISK/usr
mkdir -p $DISK/opt
mkdir -p $DISK/tmp
mkdir -p $DISK/log
mkdir -p $DISK/docker
mkdir -p $DISK/cache/passwall
mkdir -p $DISK/cache/adg
mkdir -p $DISK/cache/dnsmasq
mkdir -p $DISK/home

# 3. 设置权限
chmod 755 $DISK/overlay
chmod 755 $DISK/usr
chmod 755 $DISK/tmp
chmod 755 $DISK/log

# ============================================
# 4. 迁移软件包安装目录（关键！）
# ============================================

# 备份原 /usr
echo "=== 备份 /usr 到硬盘 ==="
cp -a /usr/* $DISK/usr/ 2>/dev/null || true

# 绑定挂载 /usr 到硬盘
mount --bind $DISK/usr /usr

# 添加到 /etc/config/fstab 永久生效
cat >> /etc/config/fstab << 'EOF'
config mount
    option target '/usr'
    option device '$DISK/usr'
    option fstype 'none'
    option options 'bind'
    option enabled '1'
EOF

# ============================================
# 5. 迁移 /tmp 缓存（关键！）
# ============================================

echo "=== 迁移 /tmp 到硬盘 ==="
# 先复制现有内容
cp -a /tmp/* $DISK/tmp/ 2>/dev/null || true

# 重新挂载 /tmp
mount --bind $DISK/tmp /tmp

# 永久配置
cat >> /etc/config/fstab << 'EOF'
config mount
    option target '/tmp'
    option device '$DISK/tmp'
    option fstype 'none'
    option options 'bind'
    option enabled '1'
EOF

# ============================================
# 6. 迁移日志目录
# ============================================

echo "=== 迁移日志到硬盘 ==="
# 创建日志链接
rm -rf /var/log 2>/dev/null
ln -sf $DISK/log /var/log

# 各软件日志目录
mkdir -p $DISK/log/passwall
mkdir -p $DISK/log/adg
mkdir -p $DISK/log/dnsmasq
mkdir -p $DISK/log/nginx

# ============================================
# 7. 配置 iStore 商店安装路径
# ============================================

echo "=== 配置 iStore 安装路径 ==="
mkdir -p $DISK/istore
uci set istore.main.path='$DISK/istore'
uci commit istore

# ============================================
# 8. 配置 Docker 根目录（如果装 Docker）
# ============================================

echo "=== 配置 Docker 根目录 ==="
uci set dockerd.globals.data_root='$DISK/docker'
uci commit dockerd

# ============================================
# 9. 配置代理缓存目录（Passwall/OpenClash）
# ============================================

echo "=== 配置代理缓存目录 ==="

# Passwall 缓存
mkdir -p $DISK/cache/passwall/{rules,tmp}
uci set passwall.@global[0].v2ray_location_asset='$DISK/cache/passwall/rules'
uci commit passwall

# OpenClash 缓存（如果用）
mkdir -p $DISK/cache/openclash
uci set openclash.config.clash_config_path='$DISK/cache/openclash'
uci commit openclash

# ============================================
# 10. 配置 AdGuardHome 工作目录
# ============================================

echo "=== 配置 AdGuardHome 工作目录 ==="
mkdir -p $DISK/cache/adg/{work,data}
uci set adguardhome.config.workdir='$DISK/cache/adg/work'
uci set adguardhome.config.datadir='$DISK/cache/adg/data'
uci commit adguardhome

# ============================================
# 11. 配置 Samba/NAS 缓存（如果用）
# ============================================

echo "=== 配置 Samba 缓存 ==="
mkdir -p $DISK/cache/samba
uci set samba4.@samba[0].socket_options='IPTOS_LOWDELAY TCP_NODELAY SO_KEEPALIVE'
uci commit samba4

# ============================================
# 12. 配置系统临时文件
# ============================================

echo "=== 配置系统临时文件 ==="

# 修改环境变量
cat >> /etc/profile << 'EOF'
export TMPDIR=/mnt/sda4/tmp
export TEMP=/mnt/sda4/tmp
export TMP=/mnt/sda4/tmp
EOF

# 修改 opkg 临时目录
mkdir -p $DISK/cache/opkg
cat > /etc/opkg.conf << 'EOF'
dest root /
dest ram /tmp
lists_dir ext /var/opkg-lists
option overlay_root /overlay
option tmp_dir /mnt/sda4/cache/opkg
EOF

# ============================================
# 13. 配置日志轮转（防止占满硬盘）
# ============================================

echo "=== 配置日志轮转 ==="
cat > /etc/logrotate.conf << 'EOF'
$DISK/log/*.log {
    daily
    rotate 7
    compress
    delaycompress
    missingok
    notifempty
    create 644 root root
}
EOF

# ============================================
# 14. 重启关键服务
# ============================================

echo "=== 重启服务 ==="
/etc/init.d/fstab enable
/etc/init.d/fstab restart
/etc/init.d/uhttpd restart
/etc/init.d/dnsmasq restart
/etc/init.d/samba4 restart 2>/dev/null || true

echo ""
echo "=========================================="
echo "  迁移完成！所有大文件已指向硬盘"
echo "  硬盘路径: $DISK"

echo ""
echo "硬盘使用情况："
